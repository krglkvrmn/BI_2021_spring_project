paralogs_cnt = []
for filename in os.listdir("ecoli_HOGs_partialid/"):
    parser = SeqIO.parse(os.path.join("ecoli_HOGs_partialid/", filename), "fasta")
    counter = Counter()
    for r in parser:
        taxid = r.description.split(" | ")[1]
        counter[taxid] += 1
    paralogs_cnt.append(sum(filter(lambda x: x > 1, counter.values())))
sns.histplot(paralogs_cnt, bins=20)
plt.title("Number of paralogs in groups with partial HOG id")
print("Groups without paralogs:", paralogs_cnt.count(0), "of", len(paralogs_cnt))

counter = Counter()
strength_counter = Counter()
avg_motifs_per_group = []
for filename in os.listdir("ecoli_HOGs_partialid/"):
    parser = SeqIO.parse(os.path.join("ecoli_HOGs_partialid/", filename), "fasta")
    l = 0
    group_motif_avg = []   
    for r in parser:
        try:
            motifs = r.description.split(" | ")[3].split()
            for motif in motifs:
                strength_counter.update(motif.split(":")[3])
            group_motif_avg.append(len(motifs))
            counter[filename] += 1
        except IndexError:
            pass
        l += 1
    counter[filename] = counter[filename] / l
    avg_motifs_per_group.append(sum(group_motif_avg)/len(group_motif_avg))


sns.histplot(list(counter.values()), bins=20)
plt.title("Proportion of proteins containing polyproline motifs in groups")
